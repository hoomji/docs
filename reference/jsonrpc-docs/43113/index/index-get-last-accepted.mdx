  <ParamField query='chainId' type='string' default='43113' required>

  <ParamField body='jsonrpc' type='string' default='2.0' required>

  </ParamField>
  <ParamField body='method' type='string' default='index.getLastAccepted' required>
    JSON-RPC method name.
  </ParamField>

  Parameters for <code>index.getLastAccepted</code>:

  </ul>
</ParamField>

### Returns

<ResponseField name='result' type='object'>
  Details of the last accepted container on the chain.
  <Expandable title='properties'>
    <ResponseField name='id' type='string'>
      The container ID of the last accepted container.
    </ResponseField>
    <ResponseField name='bytes' type='string'>
      Hex-encoded byte representation of the container.

    <ResponseField name='timestamp' type='string'>
      Unix timestamp when the container was accepted.
    </ResponseField>
    <ResponseField name='encoding' type='string'>
      Encoding format used in `bytes` (always "hex").

    <ResponseField name='index' type='string'>
      The sequential index of this container.
    </ResponseField>
  </Expandable>
</ResponseField>
