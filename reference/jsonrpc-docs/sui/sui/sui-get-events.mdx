title: 'sui_getEvents'
description: 'Return transaction events.'

### Parameters

  <ParamField query='chainId' type='string' default='sui' required>
    Unique identifier for a blockchain network.
  </ParamField>
  <ParamField body='id' type='number' default={1} required>
    Request identifier used to match responses.
  </ParamField>
  <ParamField body='jsonrpc' type='string' default='2.0' required>
    JSON-RPC version.
  </ParamField>
  <ParamField body='method' type='string' default='sui_getEvents' required>
    JSON-RPC method name.
  </ParamField>

<ParamField body='params' type='object' default={["11a72GCQ5hGNpWGh2QhQkkusTEGS6EDqifJqxr7nSYX"]} placeholder='["11a72GCQ5hGNpWGh2QhQkkusTEGS6EDqifJqxr7nSYX"]' required>
  Get all events emitted by the transaction.
  
  Positional parameters for <code>sui_getEvents</code>:
  <ul>
  <li><code>params[]</code>: The transaction digest.</li>
  </ul>
</ParamField>
