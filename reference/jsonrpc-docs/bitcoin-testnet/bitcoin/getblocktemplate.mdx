### Parameters

  <ParamField query='chainId' type='string' default='bitcoin-testnet' required>
    Unique identifier for a blockchain network.
  </ParamField>
  <ParamField body='id' type='number' default={1} required>
    Request identifier used to match responses.
  </ParamField>
  <ParamField body='jsonrpc' type='string' default='2.0' required>
    JSON-RPC version.
  </ParamField>
  <ParamField body='method' type='string' default='getblocktemplate' required>
    JSON-RPC method name.
  </ParamField>

<ParamField body='params' type='object' default={[{}]} placeholder='[{}]'>
  Parameters: [template_request?].
  
  Positional parameters for <code>getblocktemplate</code>:
  <ul>
  <li><code>params[]</code>: Optional template request specifying capabilities and rules.</li>
  </ul>
</ParamField>

### Returns

<ResponseField name='result' type='object'>
  Block template data including transactions and coinbase details.
</ResponseField>
