  <ParamField query='chainId' type='string' default='8217' required>

  </ParamField>
  <ParamField body='jsonrpc' type='string' default='2.0' required>

  </ParamField>
  <ParamField body='method' type='string' default='kaia_sendRawTransaction' required>
    JSON-RPC method name.
  </ParamField>

  </ul>
