<ParamField path='chainId' type='string' default='1' required>
  Unique identifier for a blockchain network.
</ParamField>

<ParamField body='provider' type='string'>
  Specify which provider to use. The project's provider priority list is used by
  default if no providers are selected. (Priority list can be managed on
  Uniblock's dashboard)
</ParamField>

<ParamField body='id' type='integer' default='1' required>
  An identifier established by the Client that MUST contain a String or Number.
</ParamField>

<ParamField body='jsonrpc' type='string' default='2.0' required>
  Version of the JSON-RPC protocol.
</ParamField>

<ParamField body='method' type='string' default='eth_getBlockByNumber' required>
  The name of the method to be invoked.
</ParamField>

<ParamField body='params' type='array' required>
  Parameters for eth_getBlockByNumber: [blockNumber, fullTransactions].
  <Expandable title='Array items'>
    <ParamField
      body='blockNumber'
      type='integer | string'
      default='latest'
      required
    >
      Block identifier. Can be an integer block number (minimum 0) or a string
      tag: "earliest", "latest", or "pending".
    </ParamField>
    <ParamField body='fullTransactions' type='boolean' default='true' required>
      If true, returns the full transaction objects. If false, returns only the
      transaction hashes.
    </ParamField>
  </Expandable>
</ParamField>
