### Parameters

  <ParamField query='chainId' type='string' default='bitcoin' required>
    Unique identifier for a blockchain network.
  </ParamField>
  <ParamField body='id' type='number' default={1} required>
    Request identifier used to match responses.
  </ParamField>
  <ParamField body='jsonrpc' type='string' default='2.0' required>
    JSON-RPC version.
  </ParamField>
  <ParamField body='method' type='string' default='getblockstats' required>
    JSON-RPC method name.
  </ParamField>

<ParamField body='params' type='object' default={["",[]]} placeholder='["",[]]' required>
  Parameters: [hash_or_height, metrics?].
  
  Positional parameters for <code>getblockstats</code>:
  <ul>
  <li><code>params[0]</code>: Block hash or height.</li>
  <li><code>params[1]</code>: Optional list of metrics to compute.</li>
  </ul>
</ParamField>

### Returns

<ResponseField name='result' type='object'>
  Selected block statistics (size, weight, fees, etc.).
</ResponseField>
