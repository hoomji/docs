### Parameters

Get all events emitted by the transaction.

- `params[]` (string): The transaction digest.

### Returns

<ResponseField name='result' type='object'>
  The JSON-RPC result.
  <Expandable title='properties'>
    <ResponseField name='data' type='array'>
      <Expandable title='items'>
        <ResponseField name='items[]' type='object'>
          <Expandable title='properties'>
            <ResponseField name='id' type='object'>
              <Expandable title='properties'>
                <ResponseField name='txDigest' type='string'>
                  Transaction digest
                </ResponseField>
                <ResponseField name='eventSeq' type='string'>
                  Event sequence number
                </ResponseField>
              </Expandable>
            </ResponseField>
            <ResponseField name='packageId' type='string'>
              Package ID of the event origin
            </ResponseField>
            <ResponseField name='transactionModule' type='string'>
              Move module that emitted the event
            </ResponseField>
            <ResponseField name='sender' type='string'>
              Sui address of the sender
            </ResponseField>
            <ResponseField name='type' type='string'>
              Move event type
            </ResponseField>
            <ResponseField name='parsedJson' type='object'>
              Parsed JSON representation of the event
            </ResponseField>
            <ResponseField name='bcsEncoding' type='string'>
              Encoding type (e.g., base64)
            </ResponseField>
            <ResponseField name='bcs' type='string'>
              BCS serialized event data
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name='nextCursor' type='object'>
      <Expandable title='properties'>
        <ResponseField name='txDigest' type='string'>
        </ResponseField>
        <ResponseField name='eventSeq' type='string'>
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name='hasNextPage' type='boolean'>
      True if there are more pages of events to fetch
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash Curl
curl --request POST \
  --url https://api.uniblock.dev/uni/v1/json-rpc?chainId=sui \
  --header 'content-type: application/json' \
  --data '{
    "id": 1,
    "jsonrpc": "2.0",
    "method": "sui_getEvents",
    "params": [
      "11a72GCQ5hGNpWGh2QhQkkusTEGS6EDqifJqxr7nSYX"
    ]
  }'
```
