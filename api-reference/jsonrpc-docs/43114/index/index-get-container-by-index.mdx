<ParamField body='params' type='object' default={{"index":0,"encoding":"hex"}} placeholder='{"index":0,"encoding":"hex"}' required>
  Parameters to specify the container index and encoding format.
  
  Parameters for <code>index.getContainerByIndex</code>:
  <ul>

  </ul>
</ParamField>
