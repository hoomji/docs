playground: 'none'

### Parameters

This method does not require any parameters.

### Returns

<ResponseField name='result' type='oneOf'>
  A syncing status object containing:
  <Expandable title='variants'>
    <ResponseField name='variant1' type='boolean'>
      Returns false if the node is not currently syncing state.
    </ResponseField>
    <ResponseField name='variant2' type='object'>
      Syncing status object containing progress details.
      <Expandable title='properties'>
        <ResponseField name='starting_block_hash' type='string'>
          The hash of the block from which the sync started.
        </ResponseField>
        <ResponseField name='starting_block_num' type='string'>
          The number of the block from which the sync started.
        </ResponseField>
        <ResponseField name='current_block_hash' type='string'>
          The hash of the current block being synchronized.
        </ResponseField>
        <ResponseField name='current_block_num' type='string'>
          The number of the current block being synchronized.
        </ResponseField>
        <ResponseField name='highest_block_hash' type='string'>
          The hash of the estimated highest block to be synchronized.
        </ResponseField>
        <ResponseField name='highest_block_num' type='string'>
          The number of the estimated highest block to be synchronized.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
